⚙️ Google Apps Script (Backend)

Paste this script into Google Sheets → Extensions → Apps Script. It runs on Google's servers 24/7 and sends emails with attachments automatically — even when your PC is off.

// ════════════════════════════════════════════════════════════════ // FollowUp Pro — Google Apps Script Backend // Paste in: Google Sheets → Extensions → Apps Script // ════════════════════════════════════════════════════════════════ // ── CONFIGURATION ────────────────────────────────────────────── const CONFIG = { SHEET_NAME: "FollowUps", // Your sheet tab name SENDER_NAME: "Your Company Name", // Shown as sender REPLY_TO: "[email protected]", // Reply-to address }; // ── COLUMN MAPPING (1-indexed) ───────────────────────────────── // A=1, B=2, C=3, D=4, E=5, F=6, G=7, H=8, I=9 const COL = { NAME: 1, // A - Recipient Name EMAIL: 2, // B - Email Address SUBJECT: 3, // C - Email Subject MESSAGE: 4, // D - Email Body (custom per row) DUE_DATE: 5, // E - Due Date SCHEDULE: 6, // F - daily / weekly / monthly FILE_ID: 7, // G - Google Drive File ID of attachment STATUS: 8, // H - pending / done LAST_SENT: 9, // I - Last sent timestamp (auto-updated) }; // ── MAIN TRIGGER: Run this daily via Apps Script Trigger ─────── function sendFollowUps() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(CONFIG.SHEET_NAME); if (!sheet) { Logger.log("Sheet not found: " + CONFIG.SHEET_NAME); return; } const lastRow = sheet.getLastRow(); if (lastRow < 2) return; // No data rows const data = sheet.getRange(2, 1, lastRow - 1, 9).getValues(); const today = new Date(); today.setHours(0, 0, 0, 0); let sentCount = 0; data.forEach((row, i) => { const rowNum = i + 2; const name = row[COL.NAME - 1]; const email = row[COL.EMAIL - 1]; const subject = row[COL.SUBJECT - 1]; const message = row[COL.MESSAGE - 1]; const schedule = (row[COL.SCHEDULE - 1] || "").toString().toLowerCase().trim(); const fileId = (row[COL.FILE_ID - 1] || "").toString().trim(); const status = (row[COL.STATUS - 1] || "").toString().toLowerCase().trim(); const lastSent = row[COL.LAST_SENT - 1]; // Only process PENDING rows with valid email if (status !== "pending" || !email) return; if (shouldSendToday(schedule, lastSent, today)) { const sent = sendEmail(name, email, subject, message, fileId); if (sent) { // Update Last Sent timestamp sheet.getRange(rowNum, COL.LAST_SENT).setValue(new Date()); sentCount++; Logger.log(`✅ Sent to ${name} (${email}) — ${schedule}`); } } }); Logger.log(`Done. Sent ${sentCount} email(s).`); } // ── SCHEDULE LOGIC ───────────────────────────────────────────── function shouldSendToday(schedule, lastSent, today) { if (!lastSent || lastSent === "") return true; // Never sent → send now const last = new Date(lastSent); last.setHours(0, 0, 0, 0); const diffDays = Math.floor((today - last) / (1000 * 60 * 60 * 24)); switch (schedule) { case "daily": return diffDays >= 1; case "weekly": return diffDays >= 7; case "monthly": return diffDays >= 30; default: return false; } } // ── EMAIL SENDER WITH ATTACHMENT ─────────────────────────────── function sendEmail(name, email, subject, message, fileId) { try { const htmlBody = buildHtmlEmail(name, message); const options = { htmlBody: htmlBody, name: CONFIG.SENDER_NAME, replyTo: CONFIG.REPLY_TO, }; // Attach file from Google Drive (as actual attachment, not link) if (fileId) { try { const file = DriveApp.getFileById(fileId); options.attachments = [file.getBlob()]; Logger.log(`📎 Attaching: ${file.getName()}`); } catch(e) { Logger.log(`⚠️ Could not attach file ID ${fileId}: ${e}`); } } GmailApp.sendEmail(email, subject, message, options); return true; } catch(e) { Logger.log(`❌ Failed to send to ${email}: ${e}`); return false; } } // ── HTML EMAIL TEMPLATE ──────────────────────────────────────── function buildHtmlEmail(name, message) { return ` <div style="font-family:Arial,sans-serif;max-width:600px;margin:0 auto"> <div style="background:#0a0a0f;padding:20px 30px;border-radius:10px 10px 0 0"> <h2 style="color:#e8461e;margin:0;font-size:1.4rem">Follow-Up Reminder</h2> </div> <div style="background:#f5f3ee;padding:30px;border-radius:0 0 10px 10px"> <p style="margin:0 0 16px">Dear <strong>${name}</strong>,</p> <div style="white-space:pre-line;line-height:1.7">${message}</div> <p style="margin:20px 0 0;font-size:0.85em;color:#888"> Please find the attachment included with this email. Reply to this email if you have any questions. </p> </div> </div>`; } // ── SETUP: Create Trigger (run this ONCE manually) ───────────── function setupDailyTrigger() { // Delete existing triggers first ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); // Create new daily trigger at 9:00 AM ScriptApp.newTrigger("sendFollowUps") .timeBased() .everyDays(1) .atHour(9) .create(); Logger.log("✅ Daily trigger set for 9:00 AM"); } // ── SHEET SETUP: Create headers (run ONCE) ───────────────────── function setupSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(CONFIG.SHEET_NAME); if (!sheet) sheet = ss.insertSheet(CONFIG.SHEET_NAME); const headers = [ "Name", "Email", "Subject", "Message", "Due Date", "Schedule", "Drive File ID", "Status", "Last Sent" ]; sheet.getRange(1, 1, 1, headers.length).setValues([headers]); sheet.getRange(1, 1, 1, headers.length) .setBackground("#0a0a0f") .setFontColor("#ffffff") .setFontWeight("bold"); Logger.log("✅ Sheet headers created!"); }
1
Open Google Sheets Create a new Google Sheet. Go to Extensions → Apps Script.
2
Paste the Script Delete all existing code. Click "Copy Script" above and paste it. Click 💾 Save.
3
Run setupSheet() In Apps Script, select setupSheet from dropdown → click ▶ Run. This creates the column headers automatically.
4
Upload Attachments to Google Drive Upload each file to Google Drive. Right-click → Get link → copy the File ID from the URL (the long string after /d/).
5
Share Drive Files with Script Make sure each Drive file is set to "Anyone with the link can view" OR the script account has access.
6
Add Your Data Fill in each row: Name, Email, Subject, Message, Due Date, daily/weekly/monthly, File ID, pending.
7
Set Up Auto-Trigger Select setupDailyTrigger → click ▶ Run. Google will now run sendFollowUps every day at 9 AM automatically — even when your PC is off!
8
Authorize & Done First run will ask for Gmail + Drive permissions. Accept them. Your auto email system is live! ✅
